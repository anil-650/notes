## Write a Program to add 2 no with member function in & out

```c++
/*
 * Program to add 2 no with member function in & out
 */

#include <iostream>
using namespace std;
class addition {
    int a, b, c, d;
    public:
    void add1(){
        cout << "Enter 2 numbers to add\n";
        cin >> a >> b ;
        cout << a << "+" << b << "=" << a+b << endl;
    }
    void add2();
};

void addition::add2()
{
    cout << "\nEnter another 2 numbers to add\n";
    cin >> c >> d;
    cout << c << "+" << d << "=" << c+d << endl;
};

// MAIN function

int main()
{
    addition no;
    no.add1();
    no.add2();

    return 0;
}
```
